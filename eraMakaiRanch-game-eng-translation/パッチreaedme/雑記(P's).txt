	以下のコードに変更すると標的状態でも切り替えられるはずです。
	ただしこの場合SHOPから入った場合と挙動が異なることになるので、
	あくまで参考程度に。
	
	SELECTCASE ARG:1
	CASE 0
		IF CFLAG:PREVIOUS_TARGET:表示箇所 == 0
			FINDELEMENT WORKER_LIST, PREVIOUS_TARGET
			;PRINTW
			IF RESULT == 0
				IF COUNT:3
					TARGET = TARGET_LIST:(COUNT:3 - 1)
				ELSEIF COUNT:2
					TARGET = LIVESTOCK_LIST:(COUNT:2 - 1)
				ELSE
					TARGET = WORKER_LIST:(COUNT:1 - 1)
				ENDIF
			ELSE
				TARGET = WORKER_LIST:(RESULT - 1)
			ENDIF
		ELSEIF CFLAG:PREVIOUS_TARGET:表示箇所 == 1
			FINDELEMENT LIVESTOCK_LIST, PREVIOUS_TARGET
			;PRINTW
			IF RESULT == 0
				IF COUNT:1
					TARGET = WORKER_LIST:(COUNT:1 - 1)
				ELSEIF COUNT:3
					TARGET = TARGET_LIST:(COUNT:3 - 1)
				ELSE
					TARGET = LIVESTOCK_LIST:(COUNT:2 - 1)
				ENDIF
			ELSE
				TARGET = LIVESTOCK_LIST:(RESULT - 1)
			ENDIF
		ELSEIF CFLAG:PREVIOUS_TARGET:表示箇所 == 2
			FINDELEMENT TARGET_LIST, PREVIOUS_TARGET
			;PRINTW
			IF RESULT == 0
				IF COUNT:2
					TARGET = LIVESTOCK_LIST:(COUNT:2 - 1)
				ELSEIF COUNT:1
					TARGET = WORKER_LIST:(COUNT:1 - 1)
				ELSE
					TARGET = TARGET_LIST:(COUNT:3 - 1)
				ENDIF
			ELSE
				TARGET = TARGET_LIST:(RESULT - 1)
			ENDIF
		ENDIF
		IF GROUPMATCH(CFLAG:TARGET:表示箇所, 0, 1, 3)
			CALL SCHEDULE_PLAN(0)
		ELSEIF CFLAG:TARGET:表示箇所 == 2
			CALL SCHEDULE_PLAN(1)
		ENDIF
	CASE 1
		IF CFLAG:PREVIOUS_TARGET:表示箇所 == 0
			FINDELEMENT WORKER_LIST, PREVIOUS_TARGET
			;PRINTW
			IF RESULT == (COUNT:1 - 1)
				IF COUNT:2
					TARGET = LIVESTOCK_LIST:0
				ELSEIF COUNT:3
					TARGET = TARGET_LIST:0
				ELSE
					TARGET = WORKER_LIST:0
				ENDIF
			ELSE
				TARGET = WORKER_LIST:(RESULT + 1)
			ENDIF
		ELSEIF CFLAG:PREVIOUS_TARGET:表示箇所 == 1
			FINDELEMENT LIVESTOCK_LIST, PREVIOUS_TARGET
			;PRINTW
			IF RESULT == (COUNT:2 - 1)
				IF COUNT:3
					TARGET = TARGET_LIST:0
				ELSEIF COUNT:1
					TARGET = WORKER_LIST:0
				ELSE
					TARGET = LIVESTOCK_LIST:0
				ENDIF
			ELSE
				TARGET = LIVESTOCK_LIST:(RESULT + 1)
			ENDIF
		ELSEIF CFLAG:PREVIOUS_TARGET:表示箇所 == 2
			FINDELEMENT TARGET_LIST, PREVIOUS_TARGET
			;PRINTW
			IF RESULT == (COUNT:3 - 1)
				IF COUNT:1
					TARGET = WORKER_LIST:0
				ELSEIF COUNT:2
					TARGET = LIVESTOCK_LIST:0
				ELSE
					TARGET = TARGET_LIST:0
				ENDIF
			ELSE
				TARGET = TARGET_LIST:(RESULT + 1)
			ENDIF
		ENDIF
		IF GROUPMATCH(CFLAG:TARGET:表示箇所, 0, 1, 3)
			CALL SCHEDULE_PLAN(0)
		ELSEIF CFLAG:TARGET:表示箇所 == 2
			CALL SCHEDULE_PLAN(1)
		ENDIF
	ENDSELECT
